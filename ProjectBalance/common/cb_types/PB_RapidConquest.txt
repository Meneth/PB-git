#County conquest
minor_conquest = {
	name = CB_NAME_MINOR_CONQUEST
	war_name = WAR_NAME_MINOR_CONQUEST
	sprite = 8
	truce_days = 1825
	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = no
	can_call_allies = no
	#check_de_jure_tier = COUNT # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	
	can_use = {
		has_global_flag = rapid_conquest
		ROOT = {
			NOT = { num_of_count_titles = 8 }
			OR = {
				NOT = { num_of_count_titles = 5 }
				AND = {
					capital_scope = {
						any_neighbor_province = {
							owner = {
								OR = {
									character = FROM
									is_liege_or_above = FROM
								}
							}
						}
					}
					NOT = { tier = duke }
				}
			}
			independent = yes
			OR = {
				NOT = { any_vassal = { tier = count } }
				capital_scope = {
					any_neighbor_province = {
						owner = {
							OR = {
								character = FROM
								is_liege_or_above = FROM
							}
						}
					}
				}
			}
		}
		FROM = {
			NOT = { num_of_count_titles = 3 }
			any_realm_province = {
				any_neighbor_province = {
					owner = {
						OR = {
							character = ROOT
							is_liege_or_above = ROOT
						}
					}
				}
			}
		}
	}
	
	on_success = {
		ROOT = {
			prestige = 100
		}
		FROM = {
			prestige = -100
			any_demesne_title = {
				limit = { tier = count }
				usurp_title = ROOT
			}
			any_realm_title = {
				limit = { tier = count }
				usurp_title = ROOT
			}
		}
	}
	
	on_fail = {
		ROOT = {
			prestige = -25
		}
	}
	
	on_reverse_demand = {
		ROOT = {
			prestige = -50
			transfer_scaled_wealth = {
				to = FROM
				value = 2.0
			}
		}
		FROM = {
			prestige = 50
		}
	}
	
	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}

#Subjugation
subjugation = {
	name = CB_NAME_SUBJUGATION
	war_name = WAR_NAME_SUBJUGATION
	sprite = 8
	truce_days = 1825
	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = no
	can_call_allies = no
	#check_de_jure_tier = COUNT # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	
	can_use = {
		has_global_flag = rapid_conquest
		ROOT = {
			NOT = { tier = count }
		}
		FROM = {
			OR = {
				tier = count
				NOT = { realm_size = 10 }
			}
			any_realm_province = {
				any_neighbor_province = {
					owner = {
						OR = {
							character = ROOT
							is_liege_or_above = ROOT
						}
					}
				}
			}
		}
	}
	
	on_success = {
		ROOT = {
			prestige = 100
		}
		FROM = {
			prestige = -100
			if = {
				limit = { NOT = { tier = count } }
				any_demesne_title = {
					limit = { higher_tier_than = count }
					usurp_title = ROOT
				}
			}
			set_defacto_liege = ROOT
		}
	}
	
	on_fail = {
		ROOT = {
			prestige = -25
		}
	}
	
	on_reverse_demand = {
		ROOT = {
			prestige = -50
			transfer_scaled_wealth = {
				to = FROM
				value = 2.0
			}
		}
		FROM = {
			prestige = 50
		}
	}
	
	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
