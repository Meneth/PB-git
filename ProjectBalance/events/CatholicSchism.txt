namespace = CathSchism

# This event will convert rulers from Orthodox before changing Orthodox to a heresy
# Keep that in mind when setting ai_chance modifiers and such
narrative_event = {
	id = CathSchism.100
	title = EVTNAMECathSchism.100
	desc = EVTDESCCathSchism.100
	picture = GFX_evt_cathedral
	
	is_triggered_only = yes
	major = yes
	show_ROOT = yes
	only_rulers = yes
	
	# Notify other rulers
	# These options come first so the AI has something harmless to choose as a fallback
	option = {
		trigger = {
			religion = catholic
			NOT = { character = ROOT }
		}
		name = EVTOPTALoR.200
	}
	option = {
		trigger = {
			religion_group = christian
			NOT = { religion = catholic }
			OR = {
				NOT = { religion = orthodox }
				AND = {
					religion = orthodox
					controls_religion = yes
				}
			}
		}
		name = EVTOPTA55011
	}
	option = {
		trigger = {
			NOT = { religion_group = christian }		
		}
		name = EVTOPTB86845
	}
	# The character who mended the schism
	option = {
		trigger = { character = ROOT }
		name = EVTOPTALoR.200
		set_parent_religion = { religion = orthodox parent = catholic }
		set_global_flag = schism_mended
		set_global_flag = cath_schism_mended # For eventual use with ziji's adjusted Crusades
		ROOT = { give_nickname = nick_the_saint }
	}
	# Orthodox rulers choose whether to convert
	# There should probably be some modifiers added to the ai_chance here
	option = { # Yes
		trigger = {
			religion = orthodox
			controls_religion = no
		}
		
		name = EVTOPTACathSchism.101
		religion = catholic
		any_courtier = {
			religion = catholic
		}
		capital_scope = {
			religion = catholic
		}
		ai_chance = { factor = 75 }
	}
	option = { # No
		trigger = {
			religion = orthodox
			controls_religion = no
		}
		name = EVTOPTBCathSchism.101
		ai_chance = { factor = 25 }
	}
}

#patriarch events
# Patriarchate of Antioch
character_event = {
	id = CathScism.102
	desc = EVTDESCCathSchism.102
	picture = GFX_evt_bishop
	
	trigger = {
		has_dlc = "Legacy of Rome"
		religion = catholic
		OR = {
			has_landed_title = k_papal_state
			num_of_emperor_titles = 1
		}
		NOT = { has_global_flag = pat_antioch_catholic }
		completely_controls = c_antiocheia
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = EVTOPTALoR.200
		set_global_flag = pat_antioch_catholic
		piety = 100
	}
}
